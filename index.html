Earnings Quick Plot (Python)


import pandas as pd
import matplotlib.pyplot as plt

df = pd.DataFrame({
  "Quarter": ["Q1","Q2","Q3","Q4"],
  "Revenue": [120, 128, 135, 142]
})

plt.plot(df["Quarter"], df["Revenue"], marker="o")
plt.title("Revenue by Quarter")
plt.xlabel("Quarter"); plt.ylabel("$M")
plt.show()
